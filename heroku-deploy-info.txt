Commands:

// login to heroku CLI

// make sure that Heroku Deploy CLI Plugin is installed

// create heroku app
heroku create <APP-NAME> --no-remote

// build a jar file
mvn clean package

// publish to heroku
heroku deploy:jar target/todoapp-0.0.1-SNAPSHOT.jar
